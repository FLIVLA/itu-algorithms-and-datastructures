\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{shapes,arrows,positioning}

\begin{document}
\begin{tikzpicture}[
    node distance=1cm,
    every node/.style={
        draw=white,
        circle,
        text=white % Set text color to white
    },
    every path/.style={
        draw=white,
        -latex % Set arrow style to be white
    }
]
    \node[circle,draw](1){1};
    \node[circle,draw](3) [below=of 1] {3};
    \node[circle,draw](4) [below left=of 3] {4};
    \node[circle,draw](5) [below right=of 3] {5};
    \node[circle,draw](6) [right=of 5] {6};
    \node[circle,draw](7) [above left=of 4] {7};
    \node[circle,draw](2) [below=of 7] {2};
    \node[circle,draw](8) [left=of 7] {8};
    \path (1) edge [loop above] (1);
    \path (7) edge [loop above] (7);
    \path (8) edge [loop above] (8);
    \draw[->] (3) -- (1);
    \draw[->] (4) -- (3);
    \draw[->] (5) -- (3);
    \draw[->] (6) -- (5);
    \draw[->] (2) -- (7);
